           dl_binary = case_when(is.na(dl) | dl == 5 ~ NA,

                                dz <= 2 ~ 0,
                                 TRUE ~ 1),
           co_binary = case_when(is.na(co) | co == 5 ~ NA,
                                co <= 2 ~ 0,
                                 TRUE ~ 1),
         )
